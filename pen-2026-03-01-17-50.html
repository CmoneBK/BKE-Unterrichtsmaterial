    function simulateMeasurement() {
      const steps = ['step1', 'step2', 'step3'];
      const btn = document.querySelector('button[onclick="simulateMeasurement()"]');
      btn.disabled = true;
      btn.innerText = "Messe...";
      // Reset
      steps.forEach(id => document.getElementById(id).classList.remove('active'));
      document.getElementById('finalValue').innerHTML = "<strong>Messwert</strong><br><small>(wird ermittelt)</small>";
      document.getElementById('finalValue').style.color = "var(--text)";
      // Animation
      setTimeout(() => document.getElementById('step1').classList.add('active'), 600);
      setTimeout(() => document.getElementById('step2').classList.add('active'), 1400);
      setTimeout(() => document.getElementById('step3').classList.add('active'), 2200);
      setTimeout(() => {
        steps.forEach(id => document.getElementById(id).classList.remove('active'));
        document.getElementById('finalValue').innerHTML = "<strong>Messwert</strong><br><span style='font-size: 1.2em;'>✅ 12.04 µm</span>";
        document.getElementById('finalValue').style.color = "var(--success)";
        btn.disabled = false;
        btn.innerText = "Erneut messen";
      }, 3200);
    }

    function checkStrategy() {
      const scenario = document.getElementById('scenario').value;
      const strategy = document.getElementById('strategy').value;
      const resultDiv = document.getElementById('sim-result');
      if (scenario === 'none' || strategy === 'none') {
        resultDiv.style.display = 'block';
        resultDiv.className = 'error';
        resultDiv.innerHTML = "⚠️ Bitte wähle sowohl ein Bauteil als auch einen Prüfumfang aus, um die Simulation zu starten.";
        return;
      }
      resultDiv.style.display = 'block';
      // Logik basierend auf den Texten
      if (scenario === 'mass' && strategy === 'stichprobe_random') {
        resultDiv.className = 'success';
        resultDiv.innerHTML = "✅ <strong>Sinnvolle Entscheidung.</strong> Bei der Produktion von Massenteilen wie Schrauben oder Stiften genügt die stichprobenhafte Prüfung. Alles andere wäre unwirtschaftlich.";
      } else if (scenario === 'mass' && (strategy === '100' || strategy === 'stichprobe_interval')) {
        resultDiv.className = 'error';
        resultDiv.innerHTML = "❌ <strong>Unwirtschaftlich / Falscher Ansatz.</strong> Eine 100%-Prüfung bei Massenware treibt die Kosten extrem in die Höhe. Eine einfache stichprobenhafte Prüfung ist hier das Mittel der Wahl.";
      } else if (scenario === 'safety' && strategy === '100') {
        resultDiv.className = 'success';
        resultDiv.innerHTML = "✅ <strong>Absolut notwendig.</strong> Ein Ausfall im Kraftwerksbau oder in der Chemieindustrie birgt erhebliche Sicherheitsrisiken. In solchen Fällen wird zwingend auf eine 100%-Prüfung zurückgegriffen.";
      } else if (scenario === 'safety' && (strategy === 'stichprobe_random' || strategy === 'stichprobe_interval')) {
        resultDiv.className = 'error';
        resultDiv.innerHTML = "❌ <strong>Hohes Sicherheitsrisiko!</strong> Ein defektes Großgeräteventil kann fatale Folgen haben. Stichproben genügen den Anforderungen der Risikoabwägung hier keinesfalls.";
      } else if (scenario === 'continuous' && strategy === 'stichprobe_interval') {
        resultDiv.className = 'success';
        resultDiv.innerHTML = "✅ <strong>Effizient gelöst.</strong> Bei der Herstellung von Profilen oder Bändern genügt eine punktuelle Prüfung in festen Abständen, z.B. alle 100 Meter.";
      } else if (scenario === 'continuous' && strategy === '100') {
        resultDiv.className = 'error';
        resultDiv.innerHTML = "❌ <strong>Praktisch nicht umsetzbar.</strong> Jeden Millimeter eines endlos produzierten Bandes zu prüfen ist unverhältnismäßig. Hier prüft man nach bestimmten Längenintervallen.";
      } else if (scenario === 'continuous' && strategy === 'stichprobe_random') {
        resultDiv.className = 'error';
        resultDiv.innerHTML = "❌ <strong>Zu ungenau.</strong> Bei Endlosmaterialien verlässt man sich besser auf definierte Intervalle (z.B. alle 100m) statt auf rein zufällige Stichproben, um Schwankungen im Prozess rechtzeitig zu erkennen.";
      }
    }
    // Akkordeon Logik
    function toggleAccordion(headerElement) {
      const currentItem = headerElement.parentElement;
      const isAlreadyActive = currentItem.classList.contains('active');
      // Alle anderen Items schließen
      const allItems = document.querySelectorAll('.accordion-item');
      allItems.forEach(item => {
        item.classList.remove('active');
      });
      // Wenn das angeklickte nicht offen war, wird es nun geöffnet
      if (!isAlreadyActive) {
        currentItem.classList.add('active');
      }
    }
  </script>
